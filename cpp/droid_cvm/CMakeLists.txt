# factor graph reader library
add_library(
    fg_reader
    SHARED
    ./src/libs/fg_reader.cpp
)
target_compile_options(
    fg_reader
    PRIVATE
    ${CMAKE_CXX_FLAGS}
    ${TORCH_CXX_FLAGS}
)

    fg_reader
    PRIVATE
    ./include
    ${TORCH_INCLUDE_DIRS}
)

    fg_reader
    ${TORCH_LIBRARIES}
)
